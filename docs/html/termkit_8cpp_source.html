140extern std::string center_line(std::string line) {
141 std::string result = "";
142
143 Term_size console_size = get_term_size();
144 unsigned console_middle_point = console_size.width / 2;
145 unsigned text_middle_point = line.length() / 2;
146
147 unsigned center_point = console_middle_point - text_middle_point;
148 for (unsigned i = 0; i < center_point; i++) {
149 result += " ";
150 }
151 result += line;
152 return result;
153}

155extern std::string center_text(std::string text) {
156 std::string result = "";
